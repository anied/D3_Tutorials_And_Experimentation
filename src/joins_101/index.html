Lesson #1: just attach some arbitrary data to some p's...

                
var data = [1, 2, 3, 5, 7, 11, 13];
var paragraphs = d3.select('#sectionOne')
                    .selectAll('p')
                    .data(data);

paragraphs.enter()
    .append('p')
    .text(function (d, i) {
        return 'Prime T'+i+':  '+d;
    });
